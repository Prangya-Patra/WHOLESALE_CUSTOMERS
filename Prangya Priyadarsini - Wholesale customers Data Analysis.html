from sklearn.preprocessing import StandardScaler
from sklearn.cluster import KMeans
import matplotlib.pyplot as plt
import seaborn as sns

# Standardize the data
scaler = StandardScaler()
data_scaled = scaler.fit_transform(data[['FRESH', 'MILK', 'GROCERY', 'FROZEN', 'DETERGENTS_PAPER', 'DELICASSEN']])

# Apply KMeans clustering
kmeans = KMeans(n_clusters=3, random_state=0)
data['Cluster'] = kmeans.fit_predict(data_scaled)

# Visualize the clusters
plt.figure(figsize=(10, 6))
sns.scatterplot(data=data, x='FRESH', y='GROCERY', hue='Cluster', palette='viridis')
plt.title('Customer Segments Based on Spending')
plt.show()
